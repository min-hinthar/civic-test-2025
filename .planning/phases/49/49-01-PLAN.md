Rewrite `app/error.tsx`:
```tsx
'use client';

import { useEffect } from 'react';
import { sanitizeError } from '@/lib/errorSanitizer';
import { captureError } from '@/lib/sentry';
import { SharedErrorFallback } from '@/components/ui/SharedErrorFallback';
import { useLanguage } from '@/contexts/LanguageContext';

export default function RootError({
  error,
  reset,
}: {
  error: Error & { digest?: string };
  reset: () => void;
}) {
  // Language from context (error.tsx IS inside provider tree per layout.tsx).
  // If provider itself crashes, global-error.tsx handles it with hardcoded text.
  const { showBurmese } = useLanguage();

  useEffect(() => {
    captureError(error, { source: 'error.tsx', digest: error.digest });
  }, [error]);

  const message = sanitizeError(error);

  return (
    <SharedErrorFallback
      message={message}
      showBurmese={showBurmese}
      onRetry={reset}
      onGoHome={() => { window.location.href = '/'; }}
    />
  );
}
```
