```typescript
'use client';

import { useContext } from 'react';
import { useAuth } from '@/contexts/SupabaseAuthContext';
import { useLanguage } from '@/contexts/LanguageContext';
import { useThemeContext } from '@/contexts/ThemeContext';
import { TTSContext } from '@/contexts/TTSContext';
import { useToast } from '@/components/BilingualToast';
import { useOffline } from '@/contexts/OfflineContext';
import { useSocial } from '@/contexts/SocialContext';
import { useSRS } from '@/contexts/SRSContext';
import { useUserState } from '@/contexts/StateContext';
import { useNavigation } from '@/components/navigation/NavigationProvider';

const EXPECTED_ORDER = [
  'ErrorBoundary',
  'AuthProvider',
  'LanguageProvider',
  'ThemeProvider',
  'TTSProvider',
  'ToastProvider',
  'OfflineProvider',
  'SocialProvider',
  'SRSProvider',
  'StateProvider',
  'NavigationProvider',
];

const ORDER_REFERENCE = `Expected: ${EXPECTED_ORDER.join(' > ')}
See: .claude/learnings/provider-ordering.md`;

export function ProviderOrderGuard(): null {
  if (process.env.NODE_ENV !== 'development') return null;

  // Each hook call validates its provider is mounted above this point.
  // If missing or misordered, the hook throws -- caught and warned.
  const checks: Array<{ name: string; check: () => void }> = [
    { name: 'AuthProvider', check: () => { useAuth(); } },
    { name: 'LanguageProvider', check: () => { useLanguage(); } },
    { name: 'ThemeProvider', check: () => { useThemeContext(); } },
    { name: 'TTSProvider', check: () => { useContext(TTSContext); } },
    { name: 'ToastProvider', check: () => { useToast(); } },
    { name: 'OfflineProvider', check: () => { useOffline(); } },
    { name: 'SocialProvider', check: () => { useSocial(); } },
    { name: 'SRSProvider', check: () => { useSRS(); } },
    { name: 'StateProvider', check: () => { useUserState(); } },
    { name: 'NavigationProvider', check: () => { useNavigation(); } },
  ];

  for (const { name, check } of checks) {
    try {
      check();
    } catch {
      console.warn(`[ProviderOrderGuard] ${name} missing or misordered.\n${ORDER_REFERENCE}`);
    }
  }

  return null;
}
```

IMPORTANT: The hooks must be called unconditionally (Rules of Hooks). The array iteration pattern above won't work with hooks. Instead, call each hook directly at the top level:

```typescript

  try { useAuth(); } catch { console.warn(`[ProviderOrderGuard] AuthProvider missing or misordered.\n${ORDER_REFERENCE}`); }

  try { useThemeContext(); } catch { console.warn(`[ProviderOrderGuard] ThemeProvider missing or misordered.\n${ORDER_REFERENCE}`); }
  try { useContext(TTSContext); } catch { console.warn(`[ProviderOrderGuard] TTSProvider missing or misordered.\n${ORDER_REFERENCE}`); }
  try { useToast(); } catch { console.warn(`[ProviderOrderGuard] ToastProvider missing or misordered.\n${ORDER_REFERENCE}`); }
  try { useOffline(); } catch { console.warn(`[ProviderOrderGuard] OfflineProvider missing or misordered.\n${ORDER_REFERENCE}`); }

  try { useSRS(); } catch { console.warn(`[ProviderOrderGuard] SRSProvider missing or misordered.\n${ORDER_REFERENCE}`); }
  try { useUserState(); } catch { console.warn(`[ProviderOrderGuard] StateProvider missing or misordered.\n${ORDER_REFERENCE}`); }
  try { useNavigation(); } catch { console.warn(`[ProviderOrderGuard] NavigationProvider missing or misordered.\n${ORDER_REFERENCE}`); }

  return null;

```

CRITICAL: The early return `if (process.env.NODE_ENV !== 'development') return null;` BEFORE all hook calls violates Rules of Hooks. The production guard must be inside each try-catch, OR the component must conditionally render in ClientProviders.tsx instead:
```tsx
{process.env.NODE_ENV === 'development' && <ProviderOrderGuard />}
```
This way the component never mounts in production, so hooks are never called. The component itself can then call hooks unconditionally.

NOTE: TTSContext has no exported hook. The guard must use `useContext(TTSContext)` directly. Since TTSContext is a nullable context (`createContext<TTSContextValue | null>(null)`), the check is whether the value is null rather than whether it throws. For TTSProvider, check `if (useContext(TTSContext) === null)` and warn.
