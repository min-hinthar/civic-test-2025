   Touch target audit helper function:

   ```typescript
   async function auditTouchTargets(page: Page, pageName: string) {
     const interactiveSelectors = 'button:visible, a:visible, input:visible, select:visible, [role="button"]:visible, [role="tab"]:visible';
     const elements = page.locator(interactiveSelectors);
     const count = await elements.count();

     const violations: string[] = [];
     for (let i = 0; i < count; i++) {
       const el = elements.nth(i);
       const box = await el.boundingBox();
       if (!box) continue; // off-screen or hidden
       if (box.width === 0 || box.height === 0) continue; // collapsed

       // Skip documented exceptions
       const ariaHidden = await el.getAttribute('aria-hidden');
       if (ariaHidden === 'true') continue;

       // TourTooltip exception (D-29)
       const classes = await el.getAttribute('class') || '';
       if (classes.includes('tour-tooltip') || classes.includes('joyride')) continue;

       if (box.height < 44 || box.width < 44) {
         const tag = await el.evaluate(e => e.tagName);
         const text = (await el.textContent() || '').substring(0, 30);
         violations.push(`${pageName}: ${tag} "${text}" is ${Math.round(box.width)}x${Math.round(box.height)}px`);
       }
     }

     expect(violations, `Touch target violations on ${pageName}`).toEqual([]);
   }
   ```
